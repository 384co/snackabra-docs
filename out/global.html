   [OoQD] -> '0'  
   [lIiJ] -> '1'  
   [Zz] -> '2'  
   [A] -> '4'  
   [Ss] -> '5'  
   [G] -> '6'  
   [t] -> '7'  
   [B] -> '8'  
   [gq] -> '9'  
   [C] -> 'c'  
   [Y] -> 'y'  
   [KxX] -> 'k'  
   [M] -> 'm'  
   [n] -> 'N'  
   [P] -> 'p'  
   [uvV] -> 'U'  
   [w] -> 'W'
